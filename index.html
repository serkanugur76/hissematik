export default { async fetch(request) { const url = new URL(request.url); const headers = { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*', }; if (request.method === 'OPTIONS') { return new Response(null, { headers: {...headers, 'Access-Control-Allow-Methods': 'GET', 'Access-Control-Allow-Headers': 'Content-Type'} }); } const semboller = url.searchParams.get('semboller'); const tumhisseler = url.searchParams.get('tumhisseler'); // Tüm hisseler - Bigpara sayfalı çekme if (tumhisseler === '1') { const tumHisseler = {}; const sayfalar = [1, 2, 3, 4, 5]; // 5 sayfa = ~500 hisse await Promise.all(sayfalar.map(async (sayfa) => { try { const res = await fetch( `https://bigpara.hurriyet.com.tr/api/v1/borsa/hisseler?order=HACIM&orderType=desc&pageNumber=${sayfa}&pageSize=100`, { headers: { 'User-Agent': 'Mozilla/5.0', 'Referer': 'https://bigpara.hurriyet.com.tr' } } ); const data = await res.json(); const liste = data?.data || []; liste.forEach(h => { const kod = h.KOD || h.kod; if (kod) tumHisseler[kod] = h; }); } catch(e) {} })); return new Response(JSON.stringify({ data: Object.values(tumHisseler) }), { headers }); } // Yahoo - toplu geçmiş veri if (semboller) { const liste = semboller.split(',').slice(0, 100); const sonuclar = await Promise.allSettled( liste.map(async (s) => { const res = await fetch( `https://query1.finance.yahoo.com/v8/finance/chart/${s.trim()}.IS?interval=1d&range=3mo`, { headers: { 'User-Agent': 'Mozilla/5.0' } } ); const data = await res.json(); return { sembol: s.trim(), data }; }) ); const cevap = {}; sonuclar.forEach((r, i) => { if (r.status === 'fulfilled') cevap[liste[i]] = r.value.data; }); return new Response(JSON.stringify(cevap), { headers }); } return new Response(JSON.stringify({ error: 'parametre gerekli' }), { headers }); } }
